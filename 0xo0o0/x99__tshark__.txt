*** ShellShock
	::: tshark -w Captura.cap -i tun0					"capturar el trafico del script http-shellshock.nse"

	 :: nmap  --script  http-shellshock.nse  --script-args  uri=/folderTarged/exampla.sh  -p80  $(mvnb)

	 :: tshark  -r Captura.cap  2>/dev/null     "Leer el archivo Captura.cap"

	 :: tshark  -r Captura.cap  -Y "http"  2>/dev/null     "Filtrar por http con -Y"

	 :: tshark  -r Captura.cap  -Y "http"  -Tjson  2>/dev/null     "Mostrar en formato JSON"

	 :: tshark  -r Captura.cap  -Y "http"  -Tfields -e "tcp.payload"  2>/dev/null					"obtiene el hexadecimal"

	 :: tshark  -r Captura.cap  -Y "http"  -Tfields -e "tcp.payload"  2>/dev/null | xxd -ps -r; echo   "convert hexadecimal"
